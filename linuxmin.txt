Configurar sudo gui

cd

mkdir bin

cd bin

nano gksudo

pkexec env DISPLAY=$DISPLAY XAUTHORITY=$XAUTHORITY $@

chmod 755 gksudo #para hacerlo ejecutable

lxpolkit & #ejecutar antes para que funcione o en autostart de openbox

gksudo pcmanfm #se ejecuta como root el explorador de ficheros

Ejecutar pcmanfm como root
sudo nano /usr/share/polkit-1/actions/org.freedesktop.policykit.pcmanfm.policy

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE policyconfig PUBLIC
 "-//freedesktop//DTD PolicyKit Policy Configuration 1.0//EN"
 "http://www.freedesktop.org/standards/PolicyKit/1/policyconfig.dtd">
<policyconfig>
    <action id="org.freedesktop.policykit.pkexec.pcmanfm">
    <description>Run pcmanfm program</description>
    <message>Authentication is required to run the pcmanfm</message>
    <icon_name>accessories-archiver</icon_name>
    <defaults>
        <allow_any>auth_admin</allow_any>
        <allow_inactive>auth_admin</allow_inactive>
        <allow_active>auth_admin</allow_active>
    </defaults>
    <annotate key="org.freedesktop.policykit.exec.path">/usr/bin/pcmanfm</annotate>
    <annotate key="org.freedesktop.policykit.exec.allow_gui">true</annotate>
    </action>
</policyconfig>

pkexec pcmanfm
Openbox --exit no hace un logout correcto

Activar "sessionstop_cmd" en /etc/slim.conf:

sessionstop_cmd (killall --user $USER -TERM; systemctl restart slim)
